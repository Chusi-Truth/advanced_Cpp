cmake_minimum_required(VERSION 3.12)
project(DemoProject LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 这里改为相对路径指向 hw_proj 里的 include 和 src
include_directories(${CMAKE_SOURCE_DIR}/../hw_proj/include)

file(GLOB_RECURSE SOURCES ${CMAKE_SOURCE_DIR}/../hw_proj/src/*.cpp)

add_executable(run ${SOURCES})

target_compile_options(run PRIVATE
    -ftemplate-depth=11000
    -ftemplate-backtrace-limit=0
)

# ----------todo: 调试模式以及编译提示
option(DEBUG_MODEL "Enable debug output" OFF)
message(STATUS "Debug debug model ${DEBUG_MODEL}")

if(WIN32)
    message(STATUS "Compiling on Windows system")
elseif(UNIX)
    message(STATUS "Compiling on Linux system")
else()
    message(STATUS "Compiling on unknown system")
endif()

if(DEBUG_MODEL)
    target_compile_definitions(run PRIVATE DEBUG_MODEL=1)
    message(STATUS "Debug output ENABLED for analyze.cpp")
else()
    target_compile_definitions(run PRIVATE DEBUG_MODEL=0)
    message(STATUS "Debug output DISABLED for analyze.cpp")
endif()
# ----------


set_target_properties(run PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)


message(STATUS "源文件数量: ${SOURCES}")
message(STATUS "生成可执行文件: run")
