cmake_minimum_required(VERSION 3.12)
project(DemoProject LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 这里改为相对路径指向 hw_proj 里的 include 和 src
include_directories(${CMAKE_SOURCE_DIR}/../hw_proj/include)

file(GLOB_RECURSE SOURCES ${CMAKE_SOURCE_DIR}/../hw_proj/src/*.cpp)

add_executable(run ${SOURCES})

target_compile_options(run PRIVATE
    -ftemplate-depth=11000
    -ftemplate-backtrace-limit=0
)

# ----------todo: 调试模式以及编译提示

option(ENABLE_DEBUG "Enable debug mode (print debug info from src/analyze.cpp)" OFF)
if(ENABLE_DEBUG)
    # Debug 模式：启用调试信息、定义宏 
    target_compile_options(run PRIVATE 
        -g                  
        -O0                 
        -DDEBUG_MODE        
    )
    message(STATUS "Debug mode enabled: Debug info from src/analyze.cpp will be printed.")
else()
    # Release 模式
    target_compile_options(run PRIVATE 
        -O3                
        -DNDEBUG            
    )
    message(STATUS "Release mode enabled: Debug info from src/analyze.cpp will NOT be printed.")
endif()

if(WIN32)
    message(STATUS "Building on Windows")
elseif(UNIX)
    message(STATUS "Building on Unix/Linux")
endif()
# ----------


set_target_properties(run PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)


message(STATUS "源文件数量: ${SOURCES}")
message(STATUS "生成可执行文件: run")
